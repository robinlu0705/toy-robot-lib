import Utils from './Utils';
 
function newPosCalc(x, y, f) {
  switch (f) {
    case 'NORTH': {
      return {
        x,
        y: y + 1,
      };
    }
 
    case 'EAST': {
      return {
        x: x + 1,
        y,
      };
    }
 
    case 'SOUTH': {
      return {
        x,
        y: y - 1,
      };
    }
 
    case 'WEST': {
      return {
        x: x - 1,
        y,
      };
    }
 
    default: {
      return { x, y, f };
    }
  }
}
